/**
 * @struct Graph
 * @brief Represents a graph data structure.
 *
 * The Graph struct encapsulates the graph's properties and data. It uses an
 * adjacency list representation for efficient storage and traversal.
 */
typedef struct Graph {
    int num_vertices;          /**< Number of vertices in the graph. */
    int is_directed;           /**< 1 if directed, 0 if undirected. */
    int is_weighted;           /**< 1 if edges have weights, 0 otherwise. */
    struct AdjList** adj_list; /**< Array of adjacency lists, one per vertex. */
} Graph;

/**
 * @struct AdjListNode
 * @brief Represents a node in the adjacency list.
 *
 * Each node contains the destination vertex and optionally the edge weight.
 */
typedef struct AdjListNode {
    int dest;                  /**< Destination vertex index. */
    int weight;                /**< Edge weight (ignored if graph is unweighted). */
    struct AdjListNode* next;  /**< Pointer to the next node in the list. */
} AdjListNode;

/**
 * @struct AdjList
 * @brief Represents an adjacency list for a single vertex.
 *
 * The head points to the first node in the list of adjacent vertices.
 */
typedef struct AdjList {
    AdjListNode* head;         /**< Head of the adjacency list. */
} AdjList;

/**
 * @brief Creates a new graph with the specified number of vertices.
 *
 * Allocates memory for the graph structure and initializes the adjacency lists.
 * The graph can be directed or undirected, weighted or unweighted.
 *
 * @param num_vertices Number of vertices in the graph (must be > 0).
 * @param is_directed 1 for directed graph, 0 for undirected.
 * @param is_weighted 1 for weighted graph, 0 for unweighted.
 * @return Pointer to the newly created Graph, or NULL on failure.
 *
 * @note Caller is responsible for freeing the graph using destroy_graph().
 */
Graph* create_graph(int num_vertices, int is_directed, int is_weighted);

/**
 * @brief Adds an edge between two vertices.
 *
 * For undirected graphs, adds edges in both directions. For weighted graphs,
 * uses the provided weight; otherwise, weight is ignored.
 *
 * @param graph Pointer to the Graph.
 * @param src Source vertex index (0-based).
 * @param dest Destination vertex index (0-based).
 * @param weight Edge weight (used only if graph is weighted).
 *
 * @pre src and dest must be valid vertex indices.
 * @note Does not check for duplicate edges; assumes no self-loops unless intended.
 */
void add_edge(Graph* graph, int src, int dest, int weight);

/**
 * @brief Removes an edge between two vertices.
 *
 * For undirected graphs, removes edges in both directions.
 *
 * @param graph Pointer to the Graph.
 * @param src Source vertex index.
 * @param dest Destination vertex index.
 *
 * @pre src and dest must be valid vertex indices.
 */
void remove_edge(Graph* graph, int src, int dest);

/**

 *
 * Visits all reachable vertices from the start vertex, printing them.
 * Uses a recursive approach.
 *
 * @param graph Pointer to the Graph.
 * @param start Starting vertex index.
 * @param visited Array to track visited vertices (size num_vertices).
 *
 * @pre visited array must be initialized to 0.
 */
void dfs(Graph* graph, int start, int* visited);

/**
 * @brief Performs a breadth-first search (BFS) traversal starting from a vertex.
 *
 * Visits all reachable vertices level by level, printing them.
 *
 * @param graph Pointer to the Graph.
 * @param start Starting vertex index.
 *
 * @pre start must be a valid vertex index.
 */
void bfs(Graph* graph, int start);

/**
 * @brief Prints the adjacency list representation of the graph.

 * Useful for debugging and visualization.
 *
 * @param graph Pointer to the Graph.
 */
void print_graph(Graph* graph);

/**
 * @brief Destroys the graph and frees all allocated memory.

void destroy_graph(Graph** graph);
